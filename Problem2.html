ny_noaa_sample = 
  ny_noaa |> 
  mutate(prcp = prcp/10,
         tmax = as.numeric(tmax)/10, 
         tmin = as.numeric(tmin)/10,
         snwd = as.numeric(snwd),
         year = year(date),
         month = month(date),
         day = day(date)) |> 
  select(year, month, day, everything(), -date) |>
  drop_na(prcp, snow, tmax, tmin) |>
  sample_n(5000)
